# Step 2â€“4 â€” in your agent code
from enact import EnactClient
from enact.connectors.github import GitHubConnector
from enact.connectors.postgres import PostgresConnector
from enact.connectors.filesystem import FilesystemConnector
from enact.policies.git import dont_push_to_main, dont_merge_to_main
from enact.policies.db import dont_delete_without_where
from enact.policies.filesystem import dont_delete_file, restrict_paths
from enact.workflows.agent_pr_workflow import agent_pr_workflow

enact = EnactClient(
    secret="your-hmac-secret",
    systems={
        "github":     GitHubConnector(token="...", allowed_actions=["create_branch", "create_pr"]),
        "postgres":   PostgresConnector(dsn="postgresql://..."),
        "filesystem": FilesystemConnector(base_dir="/workspace"),
    },
    policies=[
        dont_push_to_main,          # blocks direct pushes to main
        dont_merge_to_main,         # blocks PRs targeting main or master
        dont_delete_without_where,  # blocks unsafe bulk deletes
        dont_delete_file,           # blocks all file deletions
        restrict_paths(["/workspace/src"]),  # path confinement
    ],
    workflows=[agent_pr_workflow],
)

# Your agent calls this one function
result, receipt = enact.run(
    workflow="agent_pr_workflow",
    user_email="agent@company.com",
    payload={"repo": "myorg/myapp", "branch": "agent/fix-123", "title": "Fix bug"},
)

# Works with any framework that can call a Python function
